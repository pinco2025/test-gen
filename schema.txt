CREATE TABLE IF NOT EXISTS questions (
    -- Primary identifier
    uuid TEXT PRIMARY KEY,                      -- 12-character unique identifier
    
    -- Question content
    question TEXT NOT NULL,                     -- The question text
    question_image_url TEXT,                    -- URL for question image
    
    -- Option A
    option_a TEXT,                              -- Option A text
    option_a_image_url TEXT,                    -- URL for option A image
    
    -- Option B
    option_b TEXT,                              -- Option B text
    option_b_image_url TEXT,                    -- URL for option B image
    
    -- Option C
    option_c TEXT,                              -- Option C text
    option_c_image_url TEXT,                    -- URL for option C image
    
    -- Option D
    option_d TEXT,                              -- Option D text
    option_d_image_url TEXT,                    -- URL for option D image
    
    -- Answer and metadata
    answer TEXT NOT NULL,                       -- Correct answer: A, B, C, or D
    type TEXT NOT NULL,                         -- Question type: chemistry, physics, etc.
    year TEXT,                                  -- Year of question (optional)
    
    -- Tags (configurable, default is 4)
    tag_1 TEXT,                                 -- First tag
    tag_2 TEXT,                                 -- Second tag
    tag_3 TEXT,                                 -- Third tag
    tag_4 TEXT,                                 -- Fourth tag
    
    -- Timestamps
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,  -- When record was created
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,  -- When record was last updated

    -- Frequency tracking
    frequency INTEGER DEFAULT 0                      -- Tracks how many times question has been selected
);
